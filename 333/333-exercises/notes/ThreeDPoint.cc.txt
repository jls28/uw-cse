// lec 11 exercise 1

#include "ThreeDPoint.h"

// CopyFrom.
ThreeDPoint &ThreeDPoint::CopyFrom(const ThreeDPoint &py) {
  x_ = py.x_;
  y_ = py.y_;
  z_ = py.z_;
  return *this;
}

// Inner product.
double ThreeDPoint::InnerProduct(const ThreeDPoint &pt) const {
  double retval = 0.0;

  retval += (x_ * pt.x_);
  retval += (y_ * pt.y_);
  retval += (z_ * pt.z_);
  return retval;
}

// Arithmetic operators.
ThreeDPoint ThreeDPoint::operator+(const ThreeDPoint &pt) const {
  ThreeDPoint retval;

  retval.x_ = x_ + pt.x_;
  retval.y_ = y_ + pt.y_;
  retval.z_ = z_ + pt.z_;
  return retval;
}

ThreeDPoint ThreeDPoint::operator-(const ThreeDPoint &pt) const {
  ThreeDPoint retval;

  retval.x_ = x_ - pt.x_;
  retval.y_ = y_ - pt.y_;
  retval.z_ = z_ - pt.z_;
  return retval;
}

// Arithmetic + assign operators.
ThreeDPoint &ThreeDPoint::operator+=(const ThreeDPoint &pt) {
  x_ += pt.x_;
  y_ += pt.y_;
  z_ += pt.z_;
  return *this;
}

ThreeDPoint &ThreeDPoint::operator-=(const ThreeDPoint &pt) {
  x_ -= pt.x_;
  y_ -= pt.y_;
  z_ -= pt.z_;
  return *this;
}
